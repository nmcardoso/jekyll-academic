## Code Snippet

```javascript
function groupBy(items, key) {
  return items.reduce((acc, item) => {
    const group = item[key] ?? "unknown";
    if (!acc[group]) acc[group] = [];
    acc[group].push(item);
    return acc;
  }, {});
}

const records = [
  { id: 1, type: "news" },
  { id: 2, type: "post" },
  { id: 3, type: "post" }
];

console.log(groupBy(records, "type"));
```
